		startstate=(GameState) state.clone();
		// state.players.get(0);	// player1
		// state.players.get(1);	// player2

		// Each player plays a Ambassador
		for (Player player : state.players)
		{
			int i = 0;
			System.out.println(player.player_username + " is playing");
			Card c = Card.getCard(state.cards,Card.CardName.Ambassador);
			c.play(player, state);
			// Hand size is not decreased for Ambassador yet, that discard happens after the play function returns.
			// So hand size should not be less than original-2 because most player can discard here is 2.
			assertFalse( player.hand.size() < startstate.players.get(i).hand.size()-2 );
			// Hand size should go down at least 1 because player will always at least discard original reveal.
			assertFalse( player.hand.size() == startstate.players.get(i).hand.size() );
			assertFalse( player.hand.size() > startstate.players.get(i).hand.size() );
			i = 1;
		}
	}
